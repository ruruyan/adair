// Basic spam resistance:

// 1) The email address is not present in the HTML markup.

// 2) It is constructed at runtime.

// This is not perfect, but it significantly reduces casual scraping.


(function () {

const user = "delphine";

const domain = "adair-jewelry.com";

const email = user + "@" + domain;


const subject = encodeURIComponent("Co-creation with Adaïr (medical devices)");

const body = encodeURIComponent(

"Hello Adaïr Team,\n\nI’m interested in learning more about your 2026 medical device initiative.\n\nName:\nRole/Field:\nOrganization:\nBest way to reach me:\n\nThank you,\n"

);


const mailto = `mailto:${email}?subject=${subject}&body=${body}`;


const contactBtn = document.getElementById("contactBtn");

contactBtn.addEventListener("click", function () {

window.location.href = mailto;

});


const copyBtn = document.getElementById("copyBtn");

copyBtn.addEventListener("click", async function () {

try {

await navigator.clipboard.writeText(email);

copyBtn.textContent = "Copied";

setTimeout(() => (copyBtn.textContent = "Copy email"), 1500);

} catch (e) {

// Fallback if clipboard permissions are blocked

prompt("Copy the email address:", email);

}

});

})();
